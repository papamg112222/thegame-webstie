// Store positions as normalized [0,1] relative coordinates let dotPositions = []; // [{x:0.5, y:0.5}, ...] function adjustDotCount(targetCount) { const currentCount = dots.length; if (targetCount > currentCount) { const numToAdd = targetCount - currentCount; for (let i = 0; i < numToAdd; i++) { const dot = document.createElement('div'); dot.className = 'dot'; // Spawn in normalized coordinates: const posX = Math.random(); const posY = Math.random(); dotPositions.push({x: posX, y: posY}); dot.style.left = posX * (simBox.clientWidth - 10) + 'px'; dot.style.top = posY * (simBox.clientHeight - 10) + 'px'; simBox.appendChild(dot); dots.push(dot); velocities.push({ x: (Math.random() * 2 - 1), y: (Math.random() * 2 - 1) }); } } else if (targetCount < currentCount) { const numToRemove = currentCount - targetCount; for (let i = 0; i < numToRemove; i++) { dots.pop().remove(); velocities.pop(); dotPositions.pop(); } } } function updateDots() { const temp = parseInt(tempSlider.value); const mass = parseInt(massSlider.value); const speedFactor = temp / (mass / 10 + 1); dots.forEach((dot, i) => { // Update normalized position with velocity scaled by speed and container size dotPositions[i].x += velocities[i].x * speedFactor / simBox.clientWidth; dotPositions[i].y += velocities[i].y * speedFactor / simBox.clientHeight; // Bounce on edges in normalized coords if (dotPositions[i].x <= 0) { dotPositions[i].x = 0; velocities[i].x *= -1; } if (dotPositions[i].x >= 1) { dotPositions[i].x = 1; velocities[i].x *= -1; } if (dotPositions[i].y <= 0) { dotPositions[i].y = 0; velocities[i].y *= -1; } if (dotPositions[i].y >= 1) { dotPositions[i].y = 1; velocities[i].y *= -1; } // Convert normalized position to pixels dot.style.left = dotPositions[i].x * (simBox.clientWidth - 10) + 'px'; dot.style.top = dotPositions[i].y * (simBox.clientHeight - 10) + 'px'; }); requestAnimationFrame(updateDots); }
